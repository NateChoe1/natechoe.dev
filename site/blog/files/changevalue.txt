#include <stdio.h>
#include <fcntl.h>
#include <unistd.h>

int main() {
	int pid;
	void *addr;
	int newval;
	printf("Enter the pid: ");
	scanf("%d", &pid);
	printf("Enter the pointer: ");
	scanf("%p", &addr);
	printf("Enter the new value: ");
	scanf("%d", &newval);

	char path[20];
	sprintf(path, "/proc/%d/mem", pid);

	int fd = open(path, O_RDWR);
	lseek(fd, (unsigned long long) addr, SEEK_SET);
	write(fd, &newval, sizeof(newval));
}
